  32    """Function that pre-processes data set to include variables pertaining to imputation.
  33
  34    Variables added include: duplicate column of the variable to impute with suffix '_post_imputation',
  35    imputation flag, imputation type (historical, KNN), imputed value, donor user ID,
  36    donor use count, donor distance, and donor imputation class details.
  37
  38    Donor-specific variables only apply to KNN imputation.
  39
  40    Default values for these variables are:
  41        - imputation flag: 0
  42        - imputation type: '' (blank string)
  43        - imputed value: NA
  44        - donor user ID: -1
  45        - donor use count: 0
  46        - donor distance: NA
  47        - donor imputation class details: NA
  48    These variables are prefixed by the name of the variable to be imputed.
  49
  50    Parameters
  51    ----------
  52
  53    data : pd.DataFrame
  54        Pandas DataFrame of input data.
  55
  56    var_to_impute : str
  57        Name of variable to impute.
  58
  59    imputation_class_vars : list of str
  60        Names of imputation class variables.
  61
  62    knn_distance : boolean (default = True)
  63        Distance of donor.
  64
  65    Returns
  66    -------
  67
  68    data_prepped : pd.DataFrame
  69        Pandas DataFrame of input data with additional columns pertaining to imputation details.
  70    """
  71
  72    data_prepped = data.copy()
  73
  74    # Create duplicate of variable to impute
  75    var_to_impute_post_imputation_name = f"{var_to_impute}_post_imputation"
  76    data_prepped[var_to_impute_post_imputation_name] = data_prepped[var_to_impute]
  77
  78    # Create imputation flag for variable to impute.
  79    imputation_flag_col_name = f"{var_to_impute}_imputation_flag"
  80    data_prepped[imputation_flag_col_name] = 0
  81
  82    # Create column that denotes type of imputation (KNN, historical).
  83    imputation_type_col_name = f"{var_to_impute}_imputation_type"
  84    data_prepped[imputation_type_col_name] = ""
  85
  86    # Create column that contains donated, imputed value.
  87    # If imputation not required or no donor found, then imputed_value is NaN.
  88    imputed_val_col_name = f"{var_to_impute}_imputed_value"
  89    data_prepped[imputed_val_col_name] = pd.NA
  90
  91    # Create column that stores userid of donor, if applicable.
  92    # If imputation not required or no donor found, then userid = -1.
  93    donor_userid_col_name = f"{var_to_impute}_donor_userid"
  94    data_prepped[donor_userid_col_name] = -1
  95

  98    data_prepped[donor_count_col_name] = 0

 100    # Create columns for donor characteristics: values of imputation class variables.

 116def continuous_to_range(
 117    data: pd.DataFrame,
 118    column: str,
 119    range_interval_index: pd.IntervalIndex,
 120    range_label_dict: Dict,
 121) -> pd.Series:
 122    """Function that converts continuous values to bucketed ranges.
 123
 124    Parameters
 125    ----------
 126
 127    data : pd.DataFrame
 128        Pandas DataFrame of input data.
 129
 130    column : str
 131        Name of column to convert continuous values to bucketed ranges.
 132
 133    range_interval_index : pd.IntervalIndex
 134        Pandas Interval Index type that defines the buckets.
 135        Note that the left bound is exclusive, right bound is inclusive.
 136
 137    range_label_dict : dict
 138        Dictionary with Pandas Interval Indices as keys and corresponding integer labels as values.
 139
 140    Returns
 141    -------
 142
 143    continuous_to_bucketed_series: pd.Series
 144        Pandas Series of column with bucketed ranges.
 145    """
 146
 147    # returns a pandas Series that converts continuous values to buckets
 148
 149    continuous_to_bucketed_series = pd.cut(
 150        data[column],
 151        range_interval_index,
 152        labels=range(1, len(range_interval_index) + 1),
 153    )
 154    continuous_to_bucketed_series = continuous_to_bucketed_series.map(range_label_dict)
 155    return continuous_to_bucketed_series

 339def nearest_neighbour_imputation_categorical(
 340    data: pd.DataFrame,
 341    var_to_impute: str,
 342    imputation_class_vars: List[str],
 343    categorical_imputation_class_vars: List[str],
 344    missing_values_to_impute: List[int],
 345    bucket_imputation: bool,
 346    bins: Optional[pd.IntervalIndex] = None,
 347    bins_to_labels: Optional[Dict] = None,
 348    multiple_possible_class_vars: Optional[List[str]] = None,
 349    max_donor_use: int = 5,
 350    userid_col_name: str = "userid",
 351) -> pd.DataFrame:
 352    """1-Nearest Neighbour imputation function for categorical imputation class variables.
 353
 354    Parameters
 355    ----------
 356
 357    data : pd.Dataframe
 358        Pandas DataFrame of input data.
 359
 360    var_to_impute : str
 361        Name of variable to be imputed.
 362
 363    imputation_class_vars : list of strings
 364        Names of all variables used to define the imputation class.
 365
 366    categorical_imputation_class_vars : list of strings
 367        Names of categorical variables used to define the imputation class.
 368
 369    missing_values_to_impute : list of integers
 370        List of missing values to impute for.
 371
 372    bucket_imputation : bool
 373        Boolean indicator for whether to generate an additional, bucketed version of the imputed column.
 374
 375    bins : pd.IntervalIndex, optional
 376        Only required if bucket_imputation == True.
 377        Pandas Interval Index type that defines the buckets.
 378        Note that the left bound is exclusive, right bound is inclusive.
 379
 380    bins_to_labels : dict, optional
 381        Only required if bucket_imputation == True.
 382        Dictionary with Pandas Interval Indices as keys and corresponding integer labels as values.
 383
 384    multiple_possible_class_vars : list of strings, optional
 385        Names of imputation class variables that can be used in place of the other
 386        ex. either work income or business earnings can be used as an imputation class variable.
 387        Note that the function only accomodates for ONE PAIR of interchangeable
 388        imputation class variables.
 389
 390    max_donor_use : int, default = 5
 391        Maximum number of times a donor can be used.
 392
 393    userid_col_name : str, default = 'userid'
 394        Name of user ID variable.
 395
 396    Returns
 397    -------
 398
 399    data_post_imputation : pd.DataFrame
 400        Pandas DataFrame of output data, with imputations applied to variable to be imputed.
 401    """
 402
 403    # Make copy of data to implement imputations.
 404    data_post_imputation = data.copy()
 405
 406    # get imputation details and donor details' variable names
 407    imputation_flag_col_name = f"{var_to_impute}_imputation_flag"
 408    imputation_type_col_name = f"{var_to_impute}_imputation_type"
 409    post_imputation_col_name = f"{var_to_impute}_post_imputation"
 410    imputed_val_col_name = f"{var_to_impute}_imputed_value"
 411    donor_userid_col_name = f"{var_to_impute}_donor_userid"
 412    donor_count_col_name = f"{var_to_impute}_donor_count"
 413    donor_characteristics_col_names = [
 414        f"{var_to_impute}_donor_{var}" for var in imputation_class_vars
 415    ]
 416    donor_distance_col_name = f"{var_to_impute}_donor_distance"
 417
 418    # initialize donor use tracking dictionary
 419    donor_use_track_dict = {k: 0 for k in data_post_imputation[userid_col_name]}
 420
 421    # Get list of imputation class variables that are numeric.
 422    numeric_imputation_class_vars = list(
 423        set(imputation_class_vars).difference(categorical_imputation_class_vars)
 424    )
 425
 426    # Slice data by categorical imputation class variables.
 427    data_post_imputation_by_slices = {
 428        slice: sub_df.reset_index(names="original_index")
 429        for slice, sub_df in data_post_imputation.groupby(
 430            categorical_imputation_class_vars
 431        )
 432    }
 433
 434    # get column numbers for indexing use later
 435    data_columns = data_post_imputation_by_slices[
 436        [k for k in data_post_imputation_by_slices.keys()][0]
 437    ].columns
 438    imputation_flag_col_idx = data_columns.get_loc(imputation_flag_col_name)
 439    imputation_type_col_idx = data_columns.get_loc(imputation_type_col_name)
 440    post_imputation_col_idx = data_columns.get_loc(post_imputation_col_name)
 441    imputed_val_col_idx = data_columns.get_loc(imputed_val_col_name)
 442    donor_userid_col_idx = data_columns.get_loc(donor_userid_col_name)
 443    donor_count_col_idx = data_columns.get_loc(donor_count_col_name)
 444    donor_characteristics_col_idxs = [
 445        data_columns.get_loc(var) for var in donor_characteristics_col_names
 446    ]
 447    donor_distance_col_idx = data_columns.get_loc(donor_distance_col_name)
 448
 449    # initialize min-max scaler
 450    minmax_scaler = MinMaxScaler()
 451
 452    # Slice data by categorical imputation class variables.
 453    data_post_imputation_by_slices = {
 454        slice: sub_df.reset_index(names="original_index")
 455        for slice, sub_df in data_post_imputation.groupby(
 456            categorical_imputation_class_vars
 457        )
 458    }
 459
 460    # Get min-max scaled imputation class variables per slice of
 461    # categorical imputation class variables.
 462    imputation_class_vars_transformed_by_slices = {
 463        slice: minmax_scaler.fit_transform(sub_df[numeric_imputation_class_vars])
 464        for slice, sub_df in data_post_imputation_by_slices.items()
 465    }
 466
 467    # Get BallTree per slice of categorical imputation class variables.
 468    neighbour_distances_indices_by_slices = {
 469        slice: BallTree(transformed_sub_df).query(
 470            transformed_sub_df,
 471            k=transformed_sub_df.shape[0],
 472            return_distance=True,
 473            dualtree=True,
 474        )
 475        for slice, transformed_sub_df in imputation_class_vars_transformed_by_slices.items()
 476    }
 477
 478    for slice, subset_data in data_post_imputation_by_slices.items():
 479
 480        for tree_index_num, data_index_num in enumerate(subset_data.index):
 481
 482            # Search for neighbours if row has missing value
 483            if pd.notna(subset_data[post_imputation_col_name][data_index_num]) and (
 484                subset_data[post_imputation_col_name][data_index_num]
 485                in missing_values_to_impute
 486            ):
 487
 488                neighbour_rank = 0
 489                continue_neighbour_search = True
 490
 491                while continue_neighbour_search:
 492
 493                    # Determine which category of the categorical imputation class variable
 494                    # the observation belongs to in order to select correct BallTree to work with.
 495
 496                    if (
 497                        tuple(
 498                            subset_data.iloc[data_index_num][
 499                                categorical_imputation_class_vars
 500                            ]
 501                        )
 502                        == slice
 503                    ):
 504                        neighbour_distances_indices = (
 505                            neighbour_distances_indices_by_slices[slice]
 506                        )
 507
 508                        donor_idx = neighbour_distances_indices[1][
 509                            tree_index_num, neighbour_rank
 510                        ]
 511                        donor_dist = neighbour_distances_indices[0][
 512                            tree_index_num, neighbour_rank
 513                        ]
 514                        donor_userid = subset_data[userid_col_name][donor_idx]
 515
 516                        # first check: donor use does not exceed max_donor_use.
 517                        if donor_use_track_dict[donor_userid] > max_donor_use:
 518                            neighbour_rank += 1
 519
 520                        # second check: if closest donor is itself, then continue neighbour search
 521                        elif donor_idx == data_index_num:
 522                            neighbour_rank += 1
 523
 524                        # third check: if donor has missing value, then continue neighbour search
 525                        elif (
 526                            data_post_imputation_by_slices[slice][
 527                                post_imputation_col_name
 528                            ][donor_idx]
 529                            in missing_values_to_impute
 530                        ):
 531                            neighbour_rank += 1
 532
 533                        # else, donor is eligible; end neighbour search
 534                        else:
 535                            continue_neighbour_search = False
 536
 537                            # impute value into post-imputation variable
 538                            data_post_imputation_by_slices[slice].iloc[
 539                                data_index_num, post_imputation_col_idx
 540                            ] = data_post_imputation_by_slices[slice][
 541                                post_imputation_col_name
 542                            ][
 543                                donor_idx
 544                            ]
 545
 546                            # indicate imputation
 547                            data_post_imputation_by_slices[slice].iloc[
 548                                data_index_num, imputation_flag_col_idx
 549                            ] = 1
 550
 551                            # indicate type of imputation
 552                            data_post_imputation_by_slices[slice].iloc[
 553                                data_index_num, imputation_type_col_idx
 554                            ] = "KNN"
 555
 556                            # save imputed value
 557                            data_post_imputation_by_slices[slice].iloc[
 558                                data_index_num, imputed_val_col_idx
 559                            ] = data_post_imputation_by_slices[slice][
 560                                post_imputation_col_name
 561                            ][
 562                                donor_idx
 563                            ]
 564
 565                            # save donor's user id
 566                            data_post_imputation_by_slices[slice].iloc[
 567                                data_index_num, donor_userid_col_idx
 568                            ] = donor_userid
 569
 570                            # increment donor's use count in donor use tracking dictionary
 571                            donor_use_track_dict[donor_userid] += 1
 572
 573                            # save donor characteristics
 574                            for n, var in enumerate(imputation_class_vars):
 575                                data_post_imputation_by_slices[slice].iloc[
 576                                    data_index_num, donor_characteristics_col_idxs[n]
 577                                ] = data_post_imputation_by_slices[slice][var][
 578                                    donor_idx
 579                                ]
 580
 581                            # save donor distance
 582                            data_post_imputation_by_slices[slice].iloc[
 583                                data_index_num, donor_distance_col_idx
 584                            ] = donor_dist
 585
 586                            # # save sub-dataframe in dictionary tracking data post-imputation
 587                            # data_post_imputation_by_slices[slice] = subset_data
 588
 589    for slice, sub_data in data_post_imputation_by_slices.items():
 590        # reset index for subsets post-imputation
 591        sub_data = sub_data.set_index("original_index")
 592        sub_data = sub_data.rename_axis(index=None)
 593
 594        # update full data set to include post-imputation information
 595        data_post_imputation.update(sub_data)
 596
 597    # populate donor use count column
 598    for row_num, user_id in enumerate(data_post_imputation[donor_userid_col_name]):
 599        if user_id != -1:
 600            data_post_imputation.iloc[row_num, (donor_count_col_idx - 1)] = (
 601                donor_use_track_dict[user_id]
 602            )
 603
 604    return data_post_imputation
